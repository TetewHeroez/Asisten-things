              \left. \frac{dy}{dx} \right|_{(a,b)} & = -1      \\
              \frac{b}{2b - a}                     & = -1      \\
              b                                    & = -2b + a \\
              3b                                   & = a       \\

          \end{align*}

\end{enumerate}
